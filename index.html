<!DOCTYPE html>

<html lang="zh-TW">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>虛擬答案卡系統</title>

    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>

    <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>

    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>

    <script src="https://cdn.tailwindcss.com"></script>

    <style>

        .no-round { border-radius: 2px !important; }

        .columns-custom { column-gap: 2rem; }

        @media (min-width: 640px) { .columns-custom { column-count: 2; } }

        @media (min-width: 1024px) { .columns-custom { column-count: 3; } }

        @media (min-width: 1280px) { .columns-custom { column-count: 4; } }

        .break-avoid { break-inside: avoid; }

    </style>

</head>

<body class="bg-gray-100 text-gray-800 font-sans">

    <div id="root"></div>


    <script type="text/babel">

        const { useState } = React;


        function App() {

            const [step, setStep] = useState('setup');

            const [testName, setTestName] = useState('');

            const [numQuestions, setNumQuestions] = useState(50);

            const [userAnswers, setUserAnswers] = useState([]);

            const [starred, setStarred] = useState([]);

            const [correctAnswersInput, setCorrectAnswersInput] = useState('');

            const [results, setResults] = useState(null);

            const [showOnlyWrong, setShowOnlyWrong] = useState(false);


            const handleStartTest = () => {

                if (numQuestions < 1 || numQuestions > 100) return alert('題數限1-100');

                setUserAnswers(Array(Number(numQuestions)).fill(''));

                setStarred(Array(Number(numQuestions)).fill(false));

                setStep('answering');

            };


            const handleAnswerSelect = (idx, opt) => {

                const newAns = [...userAnswers];

                newAns[idx] = newAns[idx] === opt ? '' : opt;

                setUserAnswers(newAns);

            };


            const toggleStar = (idx) => {

                const newStar = [...starred];

                newStar[idx] = !newStar[idx];

                setStarred(newStar);

            };


            const handleGrade = () => {

                const cleanKey = correctAnswersInput.replace(/[^a-zA-Z]/g, '').toUpperCase();

                if (!cleanKey) return alert('請輸入答案');

                

                let correctCount = 0;

                const data = userAnswers.map((ans, idx) => {

                    const key = cleanKey[idx] || '-';

                    const isCorrect = ans === key && ans !== '';

                    if (isCorrect) correctCount++;

                    return { number: idx + 1, userAns: ans || '未填', correctAns: key, isCorrect, isStarred: starred[idx] };

                });


                setResults({ score: Math.round((correctCount/numQuestions)*100), correctCount, total: numQuestions, data });

                setStep('results');

            };


            const exportFile = () => {

                let txt = `測驗：${testName || '未命名'}\n分數：${results.score}\n\n`;

                results.data.forEach(q => {

                    txt += `${q.number}. [${q.userAns}] 正確:[${q.correctAns}] ${q.isCorrect ? 'O' : 'X'}${q.isStarred ? ' *' : ''}\n`;

                });

                const blob = new Blob([txt], { type: 'text/plain' });

                const url = URL.createObjectURL(blob);

                const a = document.createElement('a');

                a.href = url;

                a.download = '成績單.txt';

                a.click();

            };


            // Setup View

            if (step === 'setup') return (

                <div className="flex h-screen items-center justify-center p-4">

                    <div className="bg-white p-8 shadow-md w-full max-w-sm no-round border border-gray-200">

                        <h1 className="text-xl font-bold mb-6 border-b pb-2 tracking-tight">建立新測驗</h1>

                        <input type="text" placeholder="測驗名稱" className="w-full mb-4 p-2 border border-gray-300 no-round outline-none focus:border-black" onChange={e => setTestName(e.target.value)} />

                        <input type="number" placeholder="題數 (1-100)" className="w-full mb-6 p-2 border border-gray-300 no-round outline-none focus:border-black" value={numQuestions} onChange={e => setNumQuestions(e.target.value)} />

                        <button onClick={handleStartTest} className="w-full bg-black text-white p-3 font-bold no-round hover:bg-gray-800 transition-colors">開始作答</button>

                    </div>

                </div>

            );


            // Answering View

            if (step === 'answering') return (

                <div className="max-w-5xl mx-auto p-4">

                    <div className="bg-white p-4 shadow-sm border border-gray-200 sticky top-0 z-10 flex justify-between items-center mb-4 no-round">

                        <div>

                            <h2 className="font-bold">{testName || '作答中'}</h2>

                            <p className="text-xs text-gray-500">進度: {userAnswers.filter(a=>a).length}/{numQuestions}</p>

                        </div>

                        <button onClick={() => setStep('grading')} className="bg-black text-white px-6 py-2 no-round font-bold hover:bg-gray-800">交卷對答案</button>

                    </div>

                    <div className="bg-white p-6 border border-gray-200 shadow-sm no-round columns-custom">

                        {userAnswers.map((ans, i) => (

                            <div key={i} className="break-avoid flex items-center justify-between py-2 border-b border-gray-50 hover:bg-gray-50">

                                <div className="flex items-center space-x-2">

                                    <span className="font-mono text-sm font-bold w-6">{i+1}.</span>

                                    <button onClick={() => toggleStar(i)} className={`text-sm ${starred[i] ? 'text-orange-500' : 'text-gray-300'}`}>★</button>

                                </div>

                                <div className="flex space-x-1">

                                    {['A','B','C','D'].map(o => (

                                        <button key={o} onClick={() => handleAnswerSelect(i, o)} className={`w-7 h-7 text-xs font-bold border-2 no-round transition-all ${ans === o ? 'bg-black border-black text-white' : 'bg-white border-gray-300 text-gray-400 hover:border-gray-500'}`}>{o}</button>

                                    ))}

                                </div>

                            </div>

                        ))}

                    </div>

                </div>

            );


            // Grading View

            if (step === 'grading') return (

                <div className="flex h-screen items-center justify-center p-4">

                    <div className="bg-white p-8 shadow-md w-full max-w-lg no-round border border-gray-200">

                        <h2 className="font-bold mb-4">貼上正確答案</h2>

                        <textarea className="w-full h-40 p-3 border border-gray-300 no-round font-mono mb-4 focus:border-black outline-none" placeholder="例如: ABCD..." onChange={e => setCorrectAnswersInput(e.target.value)}></textarea>

                        <button onClick={handleGrade} className="w-full bg-black text-white p-3 font-bold no-round hover:bg-gray-800">開始批改</button>

                        <button onClick={() => setStep('answering')} className="w-full mt-2 text-gray-500 text-sm py-2">返回修改</button>

                    </div>

                </div>

            );


            // Results View

            if (step === 'results') return (

                <div className="max-w-4xl mx-auto p-4">

                    <div className="bg-white p-8 shadow-md border-t-4 border-black text-center mb-6 no-round">

                        <h2 className="text-4xl font-black mb-2">{results.score} 分</h2>

                        <p className="text-gray-500 mb-6">答對 {results.correctCount} / {results.total} 題</p>

                        <div className="flex justify-center space-x-4">

                            <button onClick={exportFile} className="bg-gray-100 px-4 py-2 no-round font-bold border border-gray-200 hover:bg-gray-200">匯出檔案</button>

                            <button onClick={() => location.reload()} className="bg-black text-white px-4 py-2 no-round font-bold hover:bg-gray-800">新測驗</button>

                        </div>

                    </div>

                    <div className="bg-white p-6 border border-gray-200 no-round">

                        <div className="flex justify-between items-center mb-4">

                            <h3 className="font-bold">答案詳情</h3>

                            <button onClick={() => setShowOnlyWrong(!showOnlyWrong)} className="text-sm underline">{showOnlyWrong ? '顯示全部' : '只看錯題'}</button>

                        </div>

                        <div className="grid grid-cols-2 sm:grid-cols-4 gap-3">

                            {results.data.filter(q => showOnlyWrong ? !q.isCorrect : true).map(q => (

                                <div key={q.number} className={`p-2 border no-round flex justify-between items-center ${q.isCorrect ? 'bg-green-50 border-green-100' : 'bg-red-50 border-red-100'}`}>

                                    <span className="font-bold text-xs">{q.number}.</span>

                                    <div className="text-xs">

                                        <span className={q.isCorrect ? 'text-gray-700' : 'text-red-600 font-bold'}>{q.userAns}</span>

                                        <span className="mx-1">→</span>

                                        <span className="text-green-700 font-bold">{q.correctAns}</span>

                                    </div>

                                </div>

                            ))}

                        </div>

                    </div>

                </div>

            );

        }


        const root = ReactDOM.createRoot(document.getElementById('root'));

        root.render(<App />);

    </script>

</body>

</html>
